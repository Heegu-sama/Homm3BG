% Wikilink character ° command with custom kerning for characters appearing before it
\newcommand{\wikichar}[1]{%
  \directlua{
    local s = "\luaescapestring{\detokenize{#1}}"
    local last = s:sub(-1)
    if last == "." or last == "," then
      tex.print("\\kern-0.1em")
    end
  }%
  \textcolor{\hyperlinkcolor}{\textbf{°}}%
}
